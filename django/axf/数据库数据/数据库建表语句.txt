-- 闪购--左侧类型表
drop table if exists axf_foodtypes;

id int auto_increment primary key,
typeid varchar(16) not null,
typename varchar(100) not null,
childtypenames varchar(200) not null,
typesort int default 1 not null
);

drop table if exists axf_goods;
create table axf_goods(
id int auto_increment primary key,
productid varchar(16) not null,
productimg varchar(200) not null,
productname varchar(100) not null,
productlongname varchar(200) not null,
isxf int default 1 not null,
pmdesc varchar(100) not null,

drop table if exists axf_users;
create table axf_users(
id int auto_increment primary key,
username varchar(32)  not null unique,
password varchar(255) not null,
email varchar(64) not null unique,
sex bit default 0 not null,
icon varchar(256) not null,
is_delete bit default 0 not null
);
-- 购物车
drop table if exists axf_cart;
create table axf_cart(
id int auto_increment primary key,
user_id int,
goods_id int,
c_num int default 1 not null,
is_select bit default 1 not null,
constraint fk_cart_userid foreign key(user_id) references axf_users(id),
constraint fk_cart_goodsid foreign key(goods_id) references axf_goods(id)
);





drop table if exists axf_order;
create table axf_order(
id int auto_increment primary key,
user_id int,
o_num varchar(64) not null,
o_status int default 0 not null,
o_create datetime(6) not null,
constraint fk_order_userid foreign key(user_id) references axf_users(id)
);

drop table if exists axf_order_goods;
create table axf_order_goods(
id int auto_increment primary key,
goods_id int,
order_id int,
goods_num int default 1 not null,
constraint fk_order_goods_goodsid foreign key(goods_id) references axf_goods(id),
constraint fk_order_goods_orderid foreign key(order_id) references axf_order(id)
);


drop table if exists axf_users_ticket;
create table axf_users_ticket(
user_id int,
ticket varchar(256) not null,
out_time datetime(6) not null,
constraint fk_users_ticket_userid foreign key(user_id) references axf_users(id)
);
